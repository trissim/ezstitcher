36class Step: 

37 """ 

38 A processing step in a pipeline. 

39 

40 A Step encapsulates a processing operation that can be applied to images. 

41 It mirrors the functionality of process_patterns_with_variable_components 

42 while providing a more object-oriented interface. 

43 

44 Attributes: 

45 func: The processing function(s) to apply 

46 variable_components: Components that vary across files (e.g., 'z_index', 'channel') 

47 group_by: How to group files for processing (e.g., 'channel', 'site') 

48 input_dir: The input directory 

49 output_dir: The output directory 

50 well_filter: Wells to process 

51 processing_args: Additional arguments to pass to the processing function 

52 name: Human-readable name for the step 

53 """ 

54 

55 def __init__( 

56 self, 

57 func: ProcessingFunc, 

58 variable_components: VariableComponents = None, 

59 group_by: GroupBy = None, 

60 input_dir: str = None, 

61 output_dir: str = None, 

62 well_filter: WellFilter = None, 

63 name: str = None 

64 ): 

65 """ 

66 Initialize a processing step. 

67 

68 Args: 

69 func: The processing function(s) to apply. Can be: 

70 - A single callable function 

71 - A tuple of (function, kwargs) 

72 - A list of functions or (function, kwargs) tuples 

73 - A dictionary mapping component values to functions or tuples 

74 variable_components: Components that vary across files 

75 group_by: How to group files for processing 

76 input_dir: The input directory 

77 output_dir: The output directory 

78 well_filter: Wells to process 

79 name: Human-readable name for the step 

80 """ 

81 self.func = func 

82 self.variable_components = variable_components or [] 

83 self.group_by = group_by 

84 self.input_dir = input_dir 

85 self.output_dir = output_dir 

86 self.well_filter = well_filter 

87 self.name = name or self._generate_name()

89 def _generate_name(self) -> str: 

90 """ 

91 Generate a descriptive name based on the function. 

92 

93 Returns: 

94 A human-readable name for the step 

95 """ 

96 # Helper function to get name from function or function tuple 

97 def get_func_name(f): 

98 if isinstance(f, tuple) and len(f) == 2 and callable(f[0]): 

99 return getattr(f[0], '__name__', str(f[0])) 

100 if callable(f): 

101 return getattr(f, '__name__', str(f)) 

102 return str(f) 

103 

104 # Dictionary of functions 

105 if isinstance(self.func, dict): 

106 funcs = ", ".join(f"{k}:{get_func_name(f)}" for k, f in self.func.items()) 

107 return f"ChannelMappedStep({funcs})" 

108 

109 # List of functions 

110 if isinstance(self.func, list): 

111 funcs = ", ".join(get_func_name(f) for f in self.func) 

112 return f"MultiStep({funcs})" 

113 

114 # Single function or function tuple 

115 return f"Step({get_func_name(self.func)})"

117 def process(self, context: 'ProcessingContext') -> 'ProcessingContext': 

118 """ 

119 Process the step with the given context. 

120 

121 This method applies the step's processing function to the input files 

122 and saves the results to the output directory. 

123 

124 Args: 

125 context: The processing context 

126 

127 Returns: 

128 The updated processing context 

129 """ 

130 logger.info("Processing step: %s", self.name) 

131 

132 # Get directories and microscope handler 

133 input_dir = self.input_dir 

134 output_dir = self.output_dir 

135 well_filter = self.well_filter or context.well_filter 

136 orchestrator = context.orchestrator # Required, will raise AttributeError if missing 

137 microscope_handler = orchestrator.microscope_handler 

138 

139 if not input_dir: 

140 raise ValueError("Input directory must be specified") 

141 

142 # Use ImageLocator to find the actual directory containing images 

143 # This works whether input_dir is a plate folder or a subfolder 

144 actual_input_dir = ImageLocator.find_image_directory(Path(input_dir)) 

145 logger.debug("Using actual image directory: %s", actual_input_dir) 

146 

147 # Get patterns with variable components 

148 patterns_by_well = microscope_handler.auto_detect_patterns( 

149 actual_input_dir, 

150 well_filter=well_filter, 

151 variable_components=self.variable_components 

152 ) 

153 

154 # Process each well 

155 results = {} 

156 for well, patterns in patterns_by_well.items(): 

157 if well_filter and well not in well_filter: 

158 continue 

159 

160 logger.info("Processing well: %s", well) 

161 well_results = {} 

162 

163 # Prepare patterns, functions, and args 

164 grouped_patterns, component_to_funcs, component_to_args = prepare_patterns_and_functions( 

165 patterns, self.func, component=self.group_by 

166 ) 

167 

168 # Process each component 

169 for component_value, component_patterns in grouped_patterns.items(): 

170 component_func = component_to_funcs[component_value] 

171 component_args = component_to_args[component_value] 

172 output_files = [] 

173 

174 # Process each pattern 

175 for pattern in component_patterns: 

176 # Find matching files 

177 matching_files = microscope_handler.parser.path_list_from_pattern(actual_input_dir, pattern) 

178 

179 # Load images 

180 try: 

181 images = [FileSystemManager.load_image(str(Path(actual_input_dir) / filename)) for filename in matching_files] 

182 images = [img for img in images if img is not None] 

183 except Exception as e: 

184 logger.error("Error loading images: %s", str(e)) 

185 images = [] 

186 

187 if not images: 

188 continue # Skip if no valid images found 

189 

190 # Process the images with component-specific args 

191 # Process the images 

192 images = self._apply_processing(images, func=component_func) 

193 

194 # Save images and get output files 

195 pattern_files = self._save_images(actual_input_dir, output_dir, images, matching_files) 

196 if pattern_files: 

197 output_files.extend(pattern_files) 

198 

199 # Store results for this component 

200 if output_files: 

201 well_results[component_value] = output_files 

202 

203 # Store results for this well 

204 results[well] = well_results 

205 

206 # Store results in context 

207 context.results = results 

208 return context

212 def _ensure_2d(self, img): 

213 """Ensure an image is 2D by reducing dimensions if needed.""" 

214 if not isinstance(img, np.ndarray) or img.ndim <= 2: 

215 return img 

216 

217 # Try to squeeze out singleton dimensions first 

218 squeezed = np.squeeze(img) 

219 if squeezed.ndim <= 2: 

220 return squeezed 

221 

222 # If still not 2D, take first slice until it is 

223 result = img 

224 while result.ndim > 2: 

225 result = result[0] 

226 

227 logger.warning("Reduced image dimensions from %dD to 2D", img.ndim) 

228 return result

230 def _extract_function_and_args( 

231 self, 

232 func_item: FunctionWithArgs 

233 ) -> Tuple[Callable, Dict[str, Any]]: 

234 """Extract function and arguments from a function item. 

235 

236 A function item can be either a callable or a tuple of (callable, kwargs). 

237 

238 Args: 

239 func_item: Function item to extract from 

240 

241 Returns: 

242 Tuple of (function, kwargs) 

243 """ 

244 if isinstance(func_item, tuple) and len(func_item) == 2 and callable(func_item[0]): 

245 # It's a (function, kwargs) tuple 

246 return func_item[0], func_item[1] 

247 if callable(func_item): 

248 # It's just a function, use default args 

249 return func_item, {} 

250 

251 # Invalid function item 

252 logger.warning( 

253 "Invalid function item: %s. Expected callable or (callable, kwargs) tuple.", 

254 str(func_item) 

255 ) 

256 # Return a dummy function that returns the input unchanged 

257 return lambda x, **kwargs: x, {}

259 def _apply_function_list( 

260 self, 

261 images: List[np.ndarray], 

262 function_list: List[FunctionWithArgs] 

263 ) -> List[np.ndarray]: 

264 """Apply a list of functions sequentially to images. 

265 

266 Args: 

267 images: List of images to process 

268 function_list: List of functions to apply (can include tuples of (function, kwargs)) 

269 

270 Returns: 

271 List of processed images 

272 """ 

273 processed_images = images 

274 

275 for func_item in function_list: 

276 # Extract function and args 

277 func, func_args = self._extract_function_and_args(func_item) 

278 

279 # Apply the function 

280 result = self._apply_single_function(processed_images, func, func_args) 

281 processed_images = [self._ensure_2d(img) for img in result] 

282 

283 return processed_images

287 def _apply_single_function( 

288 self, 

289 images: List[np.ndarray], 

290 func: Callable, 

291 args: Dict[str, Any] 

292 ) -> List[np.ndarray]: 

293 """Apply a single processing function with specific args. 

294 

295 Args: 

296 images: List of images to process 

297 func: Processing function to apply 

298 args: Arguments to pass to the function 

299 

300 Returns: 

301 List of processed images 

302 """ 

303 try: 

304 result = func(images, **args) 

305 

306 # Handle different return types 

307 if isinstance(result, list): 

308 return result 

309 if isinstance(result, np.ndarray): 

310 func_name = getattr(func, '__name__', 'unknown') 

311 

312 # Check if this is a 3D array (stack of images) 

313 if result.ndim >= 3: 

314 # Convert 3D+ array to list of 2D arrays 

315 logger.debug( 

316 "Function %s returned a 3D array. Converting to list of 2D arrays.", 

317 func_name 

318 ) 

319 return [result[i] for i in range(result.shape[0])] 

320 

321 # It's a single 2D image 

322 logger.warning( 

323 "Function %s returned a single image instead of a list. Wrapping it.", 

324 func_name 

325 ) 

326 return [result] 

327 

328 # Unexpected return type 

329 func_name = getattr(func, '__name__', 'unknown') 

330 result_type = type(result).__name__ 

331 logger.error( 

332 "Function %s returned an unexpected type (%s). Returning original images.", 

333 func_name, 

334 result_type 

335 ) 

336 return images 

337 

338 except Exception as e: 

339 func_name = getattr(func, '__name__', str(func)) 

340 logger.exception( 

341 "Error applying processing function %s: %s", 

342 func_name, 

343 e 

344 ) 

345 return images

347 def _apply_processing( 

348 self, 

349 images: List[np.ndarray], 

350 func: Optional[ProcessingFunc] = None 

351 ) -> List[np.ndarray]: 

352 """Apply processing function(s) to a stack (list) of images. 

353 

354 Note: This method only handles single functions or lists of functions. 

355 Dictionary mapping of functions to component values is handled by 

356 prepare_patterns_and_functions before this method is called. 

357 

358 Functions can be specified in several ways: 

359 - A single callable function 

360 - A tuple of (function, kwargs) 

361 - A list of functions or (function, kwargs) tuples 

362 

363 Args: 

364 images: List of images (numpy arrays) to process. 

365 func: Processing function(s) to apply. Defaults to self.func. 

366 

367 Returns: 

368 List of processed images, or the original list if an error occurs. 

369 """ 

370 # Handle empty input 

371 if not images: 

372 return [] 

373 

374 # Get processing function 

375 processing_func = func if func is not None else self.func 

376 

377 try: 

378 # Case 1: List of functions or function tuples 

379 if isinstance(processing_func, list): 

380 return self._apply_function_list(images, processing_func) 

381 

382 # Case 2: Single function or function tuple 

383 is_callable = callable(processing_func) 

384 is_func_tuple = isinstance(processing_func, tuple) and len(processing_func) == 2 

385 

386 if is_callable or is_func_tuple: 

387 func, args = self._extract_function_and_args(processing_func) 

388 return self._apply_single_function(images, func, args) 

389 

390 # Case 3: Invalid function 

391 logger.warning("No valid processing function provided. Returning original images.") 

392 return images 

393 

394 except Exception as e: 

395 # Try to get function name, but handle the case where processing_func might be a tuple 

396 if isinstance(processing_func, tuple) and callable(processing_func[0]): 

397 func_name = getattr(processing_func[0], '__name__', str(processing_func[0])) 

398 else: 

399 func_name = getattr(processing_func, '__name__', str(processing_func)) 

400 

401 logger.exception("Error applying processing function %s: %s", func_name, e) 

402 return images

404 def _save_images(self, input_dir, output_dir, images, filenames): 

405 """Save processed images. 

406 

407 Args: 

408 input_dir: Input directory 

409 output_dir: Output directory 

410 images: Images to save 

411 filenames: Filenames to use 

412 

413 Returns: 

414 list: Paths to saved images 

415 """ 

416 if not output_dir or not images or not filenames: 

417 return [] 

418 

419 try: 

420 # Ensure output directory exists 

421 FileSystemManager.ensure_directory(output_dir) 

422 

423 # Clean up old files if working in place 

424 if input_dir == output_dir: 

425 for filename in filenames: 

426 FileSystemManager.delete_file(Path(output_dir) / filename) 

427 

428 # Initialize output files list 

429 output_files = [] 

430 

431 # Convert to list if it's a single image 

432 if isinstance(images, np.ndarray): 

433 images = [images] 

434 filenames = [filenames[0]] 

435 

436 # Save each image 

437 for i, img in enumerate(images): 

438 if i < len(filenames): 

439 output_path = Path(output_dir) / filenames[i] 

440 FileSystemManager.save_image(str(output_path), img) 

441 output_files.append(str(output_path)) 

442 

443 return output_files 

444 

445 except Exception as e: 

446 logger.error("Error saving images: %s", str(e)) 

447 return []

554class ImageStitchingStep(Step): 

555 """ 

556 A step that stitches images using position files. 

557 

558 If input_dir is not specified, it will use the pipeline's input directory by default. 

559 If positions_dir is not specified, it will try to find a directory with "positions" in its name. 

560 """ 

561 

562 def __init__(self, name=None, input_dir=None, positions_dir=None, output_dir=None, **kwargs): 

563 """ 

564 Initialize an ImageStitchingStep. 

565 

566 Args: 

567 name (str, optional): Name of the step 

568 input_dir (str, optional): Directory containing images to stitch. 

569 If not specified, uses the pipeline's input directory. 

570 positions_dir (str, optional): Directory containing position files. 

571 If not specified, tries to find a directory with "positions" in its name. 

572 output_dir (str, optional): Directory to save stitched images 

573 **kwargs: Additional arguments for the step 

574 """ 

575 super().__init__( 

576 func=None, # ImageStitchingStep doesn't use the standard func mechanism 

577 name=name or "Image Stitching", 

578 input_dir=input_dir, 

579 output_dir=output_dir, 

580 variable_components=[], # Empty list for variable_components 

581 **kwargs 

582 ) 

583 self.positions_dir = positions_dir

585 def process(self, context): 

586 """ 

587 Stitch images using the positions file. 

588 

589 Args: 

590 context: Processing context containing orchestrator and other metadata 

591 

592 Returns: 

593 Updated context 

594 """ 

595 logger.info("Processing step: %s", self.name) 

596 

597 # Get orchestrator from context 

598 orchestrator = getattr(context, 'orchestrator', None) 

599 if not orchestrator: 

600 raise ValueError("ImageStitchingStep requires an orchestrator in the context") 

601 

602 # Get well from context 

603 well = context.well_filter[0] if context.well_filter else None 

604 if not well: 

605 raise ValueError("ImageStitchingStep requires a well filter in the context") 

606 

607 # If positions_dir is not specified, try to get it from context or find it 

608 if not self.positions_dir: 

609 # First try to get from context (set by PositionGenerationStep) 

610 self.positions_dir = getattr(context, 'positions_dir', None) 

611 

612 # If still not found, try to find at parent level of plate 

613 if not self.positions_dir and orchestrator: 

614 plate_name = orchestrator.plate_path.name 

615 parent_positions_dir = orchestrator.plate_path.parent / f"{plate_name}_positions" 

616 if parent_positions_dir.exists(): 

617 self.positions_dir = parent_positions_dir 

618 logger.info(f"Using positions directory at parent level: {self.positions_dir}") 

619 else: 

620 # Fallback to existing logic if no positions directory is found 

621 self.positions_dir = FileSystemManager.find_directory_substring_recursive( 

622 Path(self.input_dir).parent, "positions") 

623 

624 # If still not found, raise an error 

625 if not self.positions_dir: 

626 raise ValueError(f"No positions directory found for well {well}") 

627 

628 # Call the stitch_images method 

629 orchestrator.stitch_images( 

630 well=well, 

631 input_dir=self.input_dir, 

632 output_dir=self.output_dir or context.output_dir, 

633 positions_file=Path(self.positions_dir) / f"{well}.csv" 

634 ) 

635 

636 return context

641def group_patterns_by(patterns, component, microscope_handler=None): 

642 """ 

643 Group patterns by the specified component. 

644 

645 Args: 

646 patterns (list): Patterns to group 

647 Returns: 

648 dict: Dictionary mapping component values to lists of patterns 

649 """ 

650 grouped_patterns = {} 

651 for pattern in patterns: 

652 # Extract the component value from the pattern 

653 component_value = microscope_handler.parser.parse_filename(pattern)[component] 

654 if component_value not in grouped_patterns: 

655 grouped_patterns[component_value] = [] 

656 grouped_patterns[component_value].append(pattern) 

657 return grouped_patterns
